# Задание 1 Линейный алгоритм «Перевод времени»
## Ввод количества часов
hours = float(input("Введите количество часов: "))
## Перевод в минуты
minutes = hours * 60
## Вывод результата
print(f"{hours} часов = {minutes} минут")

# Задание 2 Алгоритм ветвления «Сравнение чисел»
## Ввод чисел A и B
A = float(input("Введите число A: "))
B = float(input("Введите число B: "))
## Сравнение и вывод результата
if A > B:
    print(f"Число A больше: {A} > {B}")
else:
    print(f"Число B больше: {B} > {A}")

# Задание 4 Алгоритм с ветвлением «Скидка в магазине»
## Ввод цены за единицу и количества
price_per_unit = float(input("Цена за единицу (руб.): "))
quantity = float(input("Количество единиц: "))

## Расчет общей суммы
total_amount = price_per_unit * quantity

## Расчет скидки и итоговой суммы
if total_amount > 1000:
    discount = 0.1 * total_amount
    final_amount = total_amount - discount
else:
    final_amount = total_amount
## Вывод результата
print(f"Итоговая сумма к оплате: {final_amount:.2f} руб.")

# Задание 5 Алгоритм с вложенными ветвлениями
# Ввод коэффициентов a, b, c
a = float(input("Введите коэффициент a: "))
b = float(input("Введите коэффициент b: "))
c = float(input("Введите коэффициент c: "))

## Проверка типа уравнения
if a == 0:
    print("Уравнение не является квадратным")
else:
    # Вычисление дискриминанта
    D = b**2 - 4*a*c  
    # Анализ корней
    if D < 0:
        print("Корней нет")
    elif D == 0:
        x = -b / (2*a)
        print(f"Один корень: {x}")
    else:
        x1 = (-b + sqrt(D)) / (2*a)
        x2 = (-b - sqrt(D)) / (2*a)
        print(f"Корни: x1 = {x1}, x2 = {x2}")
